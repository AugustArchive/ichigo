:love_letter: | Discord RPC client made in TypeScript, based on discord-Âµrpc

Documentation | NPM

Example

 
import { Ichigo } from '@augu/ichigo';
const rpc = new Ichigo('');

rpc.on('open', () => console.log('[Ichigo] Opened connection.'));
rpc.on('error', (error) => console.error('[Ichigo] Unknown error!', error));
rpc.on('ready', () => {
    console.log('[Ichigo] Ready!');
    rpc.setActivity({
        instance: false,
        state: 'State',
        details: 'Details',
        timestamps: {
            start: new Date().getTime()
        }
    });
});

rpc.connect();
